class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* cur = head;
        ListNode* res = nullptr;
        

        while(cur)
        {
            ListNode* temp = cur->next;
            temp = cur->next;
            cur->next=res;
            res = cur;
            cur = temp;
        }
        return res;
    }
};
