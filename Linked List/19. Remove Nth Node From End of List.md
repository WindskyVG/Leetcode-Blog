class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        
        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;
        ListNode* temp = dummyHead;
        ListNode* cur = dummyHead;
        while(n--&& cur != NULL)
        {
            cur=cur->next;
        }
        cur=cur->next;
        while(cur!=nullptr)
        {
            cur=cur->next;
            temp=temp->next;
        }
        temp->next = temp->next->next;
        return dummyHead->next;
    }
};
