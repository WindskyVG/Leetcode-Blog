class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        int n = 0;
        ListNode* tmp = head;
        while (tmp != nullptr) {
            n++;
            tmp = tmp->next;
        }

        ListNode* dummy = new ListNode(0);
        dummy->next = head;
        ListNode* p = dummy;
        ListNode* cur = head;
        ListNode* pre = nullptr;

        for (; n >= k; n -= k) {
            for (int i = 0; i < k; i++) {
                ListNode* next = cur->next;
                cur->next = pre;
                pre = cur;
                cur = next;
            }

            ListNode* next = p->next;
            p->next->next = cur;
            p->next = pre;
            p = next;
        }
        return dummy->next;
    }
};
