### **Code Implementation**
```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int result=0;
        int temp = 0;
        set <int> seen;
        int i = 0;
        int j = 0;
        while(i<s.size())
        {
            if (seen.find(s[i]) != seen.end()) {
                seen.erase(s[j]);
                j++;
                continue;
            } else {
                seen.insert(s[i]);
                i++;
                temp = i-j;
                result = result > temp ? result : temp;
            }
        }
        return result;
    }
};
```
