### **Code Implementation**
```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        if(n<=1) return 0;

        int maxprofit = 0;
        int minprice = prices[0];
        for(int i = 0; i<n;i++)
        {
            minprice = (minprice<prices[i]) ? minprice : prices[i];

            maxprofit = (maxprofit<prices[i]-minprice) ? prices[i]-minprice : maxprofit;
        }
        return maxprofit;
    }
};
```
